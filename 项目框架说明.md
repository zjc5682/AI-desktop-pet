```typescript
  // 1. 创建 PIXI 应用
  app = new PIXI.Application({
    view: canvas.value,
    backgroundAlpha: 0,  // 透明背景
    width: props.width,
    height: props.height
  });
  
  // 2. 加载模型

  // 3. 添加到舞台
  app.stage.addChild(model);
  
  // 4. 调整模型到画布中心（关键！）
  resizeModel();
  
  // 5. 播放待机动画
  model.motion(model.motions.idle[0]);
});

// 调整模型大小和位置（居中显示）
const resizeModel = () => {
  if (!model || !app) return;
  
  const baseHeight = 1080;
  let scale = (props.height / baseHeight) * 0.5;
  scale = Math.max(0.2, Math.min(0.8, scale));
  
  model.scale.set(scale);
  model.anchor.set(0.5, 0.5);   // 锚点：正中心
  model.x = props.width / 2;    // 水平居中
  model.y = props.height / 2;   // 垂直居中
};
```

#### useIdleDetection.ts - 瞌睡虫检测器

```typescript
export function useIdleDetection(onIdle, onSleep) {
  let idleTimer, sleepTimeout;
  
  const resetIdleTimer = () => {
    // 5 秒后无聊
    idleTimer = setTimeout(() => {
      onIdle();  // 显示"好无聊啊~"
      
      // 再等 30 秒睡觉
      sleepTimeout = setTimeout(() => {
        isSleeping.value = true;
        onSleep();  // 播放睡眠动画
      }, 30000);
    }, 5000);
  };
  
  return { isSleeping, startIdleDetection, toggleSleep, ... };
}
```

```typescript
const { isSleeping, startIdleDetection } = useIdleDetection(
  () => showRandom(idleMessages),  // 无聊时
```

#### useHitDetection.ts - 碰撞检测员

```typescript
export function useHitDetection() {
  // 坐标转换：屏幕坐标 → Live2D 坐标
  const getNormalizedCoords = (clientX, clientY, rect) => {
    const x = ((clientX - rect.left) / rect.width) * 2 - 1;
    const y = -((clientY - rect.top) / rect.height) * 2 + 1;
    return { x, y };  // 范围：-1 ~ 1
  };
  
  // 矩形碰撞检测
  const isInHitArea = (x, y, area) => {
    return x >= area.x[0] && x <= area.x[1] &&
           y >= area.y[0] && y <= area.y[1];
  };
  
  return { isInHitArea, getNormalizedCoords };
}
```

### 8️⃣ **src-tauri/src/lib.rs** - Rust 后端窗口管理

**核心作用**：配置窗口的物理属性（边框、阴影等）

```rust
use tauri::Manager; // 导入 Manager trait

#[cfg_attr(mobile, tauri::mobile_entry_point)]
pub fn run() {
    tauri::Builder::default()
        .plugin(tauri_plugin_opener::init())
        .invoke_handler(tauri::generate_handler![greet])
        .setup(|app| {
            // 获取主窗口并设置无边框和阴影
            if let Some(window) = app.webview_windows().get("main") {
                // 设置窗口为无边框模式（移除边框）
                window.set_decorations(false).unwrap();
                
                // Windows 特定：禁用阴影效果
                #[cfg(target_os = "windows")]
                {
                    let _ = window.set_shadow(false);
                }
            }
            Ok(())
        })
        .run(tauri::generate_context!())
        .expect("error while running tauri application");
}
```

**关键点**：
- ✅ 必须导入 `use tauri::Manager;`
- ✅ 使用 `app.webview_windows().get("main")` 获取窗口
- ✅ `set_decorations(false)` 移除边框和标题栏
- ✅ `set_shadow(false)` 禁用 Windows 阴影
- ✅ 修改后需重新编译：`npm run tauri dev` 或 `npm run tauri build`

## 💡 关键设计亮点

### 1. **分层架构**
```
表现层（Components） ↔ 逻辑层（Composables） ↔ 数据层（Model）
```

### 2. **关注点分离**
- `Live2DCanvas`：只负责渲染
- `useIdleDetection`：只负责状态检测
- `useMessageSystem`：只负责消息管理

### 3. **响应式设计**
- 所有状态使用 `ref()` 和 `computed()`
- 自动追踪依赖，按需更新

### 4. **容错处理**
- 多层 API 兼容性尝试
- 备选动画列表
- 完善的错误捕获

### 5. **无边框透明窗口实现** ✨
**三层保障机制**：
1. **Tauri 配置层**：`tauri.conf.json` 设置 `"transparent": true` 和 `"decorations": false`
2. **Rust 后端层**：在 `lib.rs` 中调用 `window.set_decorations(false)` 和 `window.set_shadow(false)`
3. **前端样式层**：所有容器统一设置透明背景和无边框样式

**模型居中显示**：
- 锚点设置为 `(0.5, 0.5)`（正中心）
- X 坐标设为 `width / 2`（水平居中）
- Y 坐标设为 `height / 2`（垂直居中）
